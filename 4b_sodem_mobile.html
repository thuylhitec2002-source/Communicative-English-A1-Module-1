const numberData = { 0: ["Zero", "/ˈzɪərəʊ/"], 1: ["One", "/wʌn/"], 2: ["Two", "/tuː/"], 3: ["Three", "/θriː/"], 4: ["Four", "/fɔː/"], 5: ["Five", "/faɪv/"], 6: ["Six", "/sɪks/"], 7: ["Seven", "/ˈsev.ən/"], 8: ["Eight", "/eɪt/"], 9: ["Nine", "/naɪn/"], 10: ["Ten", "/ten/"], 11: ["Eleven", "/ɪˈlev.ən/"], 12: ["Twelve", "/twelv/"], 13: ["Thirteen", "/ˌθɜːˈtiːn/"], 14: ["Fourteen", "/ˌfɔːˈtiːn/"], 15: ["Fifteen", "/ˌfɪfˈtiːn/"], 16: ["Sixteen", "/ˌsɪkˈstiːn/"], 17: ["Seventeen", "/ˌsev.ənˈtiːn/"], 18: ["Eighteen", "/ˌeɪˈtiːn/"], 19: ["Nineteen", "/ˌnaɪnˈtiːn/"], 20: ["Twenty", "/ˈtwen.ti/"] }; let questions = []; let currentIdx = 0; let score = 0; let targetNum = null; let canClick = true; let hintUsed = false; let sndCorrect, sndWrong; // Khởi tạo audio sau khi DOM ready document.addEventListener('DOMContentLoaded', () => { sndCorrect = document.getElementById('snd-correct'); sndWrong = document.getElementById('snd-wrong'); }); // Hàm phát âm thanh an toàn function playSound(sound) { if (!sound) return; sound.currentTime = 0; sound.play().catch(() => {}); } // Unlock audio cho mobile function unlockAudio() { // Unlock speech synthesis if ('speechSynthesis' in window) { const unlockUtterance = new SpeechSynthesisUtterance(''); unlockUtterance.volume = 0; window.speechSynthesis.speak(unlockUtterance); } // Unlock audio elements [sndCorrect, sndWrong].forEach(snd => { if (snd) { snd.play().then(() => { snd.pause(); snd.currentTime = 0; }).catch(() => {}); } }); } // Reset tất cả biến về trạng thái ban đầu function resetGame() { questions = []; currentIdx = 0; score = 0; targetNum = null; canClick = true; hintUsed = false; } window.startGame = function() { unlockAudio(); // Reset game state resetGame(); // Ẩn/hiện màn hình document.getElementById('welcome-screen').style.display = 'none'; document.getElementById('result-screen').style.display = 'none'; document.getElementById('main-game').style.display = 'block'; // Tạo danh sách câu hỏi ngẫu nhiên questions = Object.keys(numberData) .map(n => parseInt(n)) .sort(() => Math.random() - 0.5); // Tạo grid buttons const grid = document.getElementById('grid'); grid.innerHTML = ''; for (let i = 0; i <= 20; i++) { const b = document.createElement('button'); b.className = 'num-btn'; b.id = 'btn-' + i; b.innerText = i; b.onclick = () => check(i, b); grid.appendChild(b); } load(); }; // Chơi lại từ đầu window.restartGame = function() { startGame(); }; function load() { if (currentIdx >= questions.length) { finish(); return; } canClick = true; hintUsed = false; targetNum = questions[currentIdx]; document.getElementById('word-text').innerText = numberData[targetNum][0]; document.getElementById('phonetic-text').innerText = numberData[targetNum][1]; document.getElementById('progress-text').innerText = `Câu ${currentIdx + 1}/${questions.length}`; document.getElementById('progress-fill').style.width = `${((currentIdx) / questions.length) * 100}%`; // Reset trạng thái các button document.querySelectorAll('.num-btn').forEach(b => { b.classList.remove('correct', 'wrong', 'highlight'); b.disabled = false; }); setTimeout(playCurrent, 120); } window.playCurrent = function() { if (!('speechSynthesis' in window)) return; // Cancel bất kỳ speech nào đang chạy if (window.speechSynthesis.speaking || window.speechSynthesis.pending) { window.speechSynthesis.cancel(); } // Delay nhỏ sau khi cancel để tránh bug trên một số trình duyệt setTimeout(() => { const utter = new SpeechSynthesisUtterance(numberData[targetNum][0]); utter.lang = 'en-US'; utter.rate = 0.85; window.speechSynthesis.speak(utter); }, 50); }; window.showHint = function() { if (!canClick) return; hintUsed = true; const b = document.getElementById('btn-' + targetNum); if (!b) return; b.classList.add('highlight'); setTimeout(() => { b.classList.remove('highlight'); }, 1000); }; function check(n, b) { if (!canClick) return; if (n === targetNum) { canClick = false; if (!hintUsed) score++; b.classList.add('correct'); playSound(sndCorrect); setTimeout(() => { currentIdx++; load(); }, 800); } else { b.classList.add('wrong'); playSound(sndWrong); setTimeout(() => { b.classList.remove('wrong'); }, 400); } } function finish() { document.getElementById('main-game').style.display = 'none'; document.getElementById('result-screen').style.display = 'block'; const total = questions.length; document.getElementById('res-score').innerText = `${score}/${total}`; let msg; const percent = score / total; if (percent === 1) { msg = "Tuyệt vời! Bạn là thiên tài số học! 🎉"; } else if (percent >= 0.8) { msg = "Xuất sắc! Bạn gần như hoàn hảo! ⭐"; } else if (percent >= 0.5) { msg = "Khá lắm! Cố gắng thêm nhé! 💪"; } else { msg = "Hãy luyện tập thêm nhé! 📚"; } document.getElementById('res-msg').innerText = msg; }
