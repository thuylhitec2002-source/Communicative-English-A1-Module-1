function playCurrentNumber() { if (targetNum === null) return; const synth = window.speechSynthesis; // Dừng mọi âm thanh đang phát để tránh bị lặp/kẹt số cũ synth.cancel(); // Lấy chữ tiếng Anh (ví dụ: "Eight") thay vì chỉ truyền số 8 const textToRead = numberData[targetNum][0]; const utter = new SpeechSynthesisUtterance(textToRead); utter.lang = 'en-US'; utter.rate = 0.8; // Giảm tốc độ một chút để nghe rõ hơn utter.pitch = 1; synth.speak(utter); }
